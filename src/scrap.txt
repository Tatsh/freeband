  /* Main event handler */
  SDL_Event freeband;
  bool gameOver = false;
  bool windowActive = true;
  
  while (!gameOver) {

    while (SDL_PollEvent(&freeband)) {

      switch (freeband.type) {

        case SDL_QUIT: /* Handle quit requests */
          gameOver = true;
#ifdef __DEBUG__
          printf("Quitting...\n");
#endif
          break;

        case SDL_ACTIVEEVENT: /* Window handling */
          if (freeband.active.gain == 0) /* We have lost focus, do not draw anymore */
            windowActive = false;
          else
            windowActive = true;
          break;

  case SDL_KEYDOWN:
          menuUpDown(&freeband.key.keysym); /* Handle key presses */
          break;

  default:
          mainMenuOption = getMainMenuOption(); /* screens/start.c */
}
      
}          
if (mainMenuOptions == 0) {
#ifdef __DEBUG__
            printf("Nothing has been selected yet. Pressing return will start a single player game.\n");
#endif
            startGame(1);
          if (mainMenuOption == 1) { /* Single player mode */
#ifdef __DEBUG__
            printf("Starting single player game.\n");
#endif
            startGame(1);
          }
          else if (mainMenuOption == 2) { /* Multiplayer mode */
#ifdef __DEBUG__
            printf("Starting multiplayer game.\n");
#endif
            startGame(2);
          }
          else if (mainMenuOption == 3) { /* Online menu */
#ifdef __DEBUG__
            printf("Displaying online menu.\n");
#endif
#ifdef __GL__
            goOnline_GL();
#endif
          }
          else if (mainMenuOption == 4) { /* Options menu */
#ifdef __DEBUG__
            printf("Displaying options menu.\n");
#endif
#ifdef __GL__
            showOptionsMenu_GL();
#endif
          }
          else
#ifdef __DEBUG__
            printf("Quitting...\n");
#endif
          break;
      }
